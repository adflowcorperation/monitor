# Exception track Rule
RULE be caught exception when connectting
CLASS javax.servlet.http.HttpServlet
METHOD service(javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
HELPER kr.co.adflow.monitor.exception.InterceptException
AT ENTRY
BIND hServlet:HttpServlet= $0,
	 request:HttpServletRequest = $1,
	 response:HttpServletResponse = $2,
	 threadId:String = Long.toString(Thread.currentThread().getId())
IF TRUE
DO throw new InterceptException(threadId)
ENDRULE
